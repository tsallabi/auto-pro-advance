// Key Feature: Triple Invoice Creation on Win
const createWinInvoices = (userId: string, carId: string, amount: number) => {
  const now = new Date().toISOString();
  const dueDate7 = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString();
  const commission = amount * 0.07; // 7% platform commission
  const transportFee = 450; // Fixed internal transport fee
  const shippingFee = 1200; // Estimated international shipping
  
  // Invoice 1: Purchase (active immediately)
  db.prepare(`INSERT INTO invoices(id, userId, carId, amount, status, type, timestamp, dueDate) VALUES(?, ?, ?, ?, 'unpaid', 'purchase', ?, ?)`)
    .run(`inv-pur-${Date.now()}`, userId, carId, amount + commission, now, dueDate7);
  // Invoice 2: Transport (pending)
  db.prepare(`INSERT INTO invoices(id, userId, carId, amount, status, type, timestamp, dueDate) VALUES(?, ?, ?, ?, 'pending', 'transport', ?, ?)`)
    .run(`inv-trn-${Date.now()}`, userId, carId, transportFee, now, dueDate7);
  // Invoice 3: Shipping (pending)
  db.prepare(`INSERT INTO invoices(id, userId, carId, amount, status, type, timestamp, dueDate) VALUES(?, ?, ?, ?, 'pending', 'shipping', ?, ?)`)
    .run(`inv-shp-${Date.now()}`, userId, carId, shippingFee, now, dueDate7);
  // ... and create shipment record.
}
